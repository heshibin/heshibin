 <properties>        <!--JDK版本，如果是jdk8则这里是 1.8-->        <java.version>11</java.version>        <maven.compiler.source>11</maven.compiler.source>        <maven.compiler.target>11</maven.compiler.target>        <spring.boot.version>2.5.5</spring.boot.version>        <mybatisplus.boot.starter.version>3.4.0</mybatisplus.boot.starter.version>        <lombok.version>1.18.16</lombok.version>        <sharding-jdbc.version>4.1.1</sharding-jdbc.version>        <junit.version>4.12</junit.version>        <druid.version>1.1.16</druid.version>        <!--跳过单元测试-->        <skipTests>true</skipTests>    </properties>

        <dependency>            <groupId>org.springframework.boot</groupId>            <artifactId>spring-boot-starter-web</artifactId>            <version>${spring.boot.version}</version>        </dependency>
